--- (2.1) AVG AGE PER POLITICAL ORIENTATION
/*This query computes the average age of users for each political orientation.
It joins the users table with the bridge table user_political_orientations and 
then with political_orientations to retrieve the category name. The HAVING clause keeps only political orientations with more than 50 users, so the results are based on categories with enough data.*/
SELECT 
	po.name as "Political Orientation", 
	ROUND(AVG(u.age)) AS "Average User Age",
	COUNT(DISTINCT(upo.user_id)) AS "User Count"
FROM 
	users AS u
	JOIN user_political_orientations AS upo ON 
		upo.user_id = u.user_id
	JOIN political_orientations AS po ON 
		upo.political_orientation_id = po.political_orientation_id
GROUP BY 
	po.name
HAVING 
	COUNT(DISTINCT(upo.user_id)) > 50
ORDER BY 
	"Average User Age";

--- (2.2) MEMBER COUNT PER GROUP
/*It counts how many distinct users belong to each group. It joins user_groups
with groups in order to retrieve the group names. The result is grouped by group name and ordered from 
the largest group to the smallest one
*/
SELECT 
	groups.name as "Group Name", 
	COUNT(DISTINCT(user_groups.user_id)) as "Member Count"
FROM 
	user_groups 
JOIN 
	"groups" ON 
	user_groups.group_id = groups.group_id
GROUP BY 
	groups.name
ORDER BY 
	"Member Count" DESC;

--- (2.3) AVG AGE PER VACCINATION STANCE
/*This third query compared vaccination stances by gender. For each vaccination stance, it calculates the average age and number of male and female users. The query uses CTEs to separetely compute male and female aggregates
and then joins them with vaccination_stances to show the final result*/
WITH male_users AS (
		SELECT 
			ROUND(AVG(u.age)) AS "age",
			vs.vaccination_stance_id,
			COUNT(DISTINCT(uvs.user_id)) AS "user_count"
		FROM
			users AS u
		JOIN
			user_vaccination_stances AS uvs ON
			uvs.user_id = u.user_id
		JOIN
			vaccination_stances AS vs ON
			vs.vaccination_stance_id = uvs.vaccination_stance_id
		WHERE
			u.gender = 'Man'
		GROUP BY
			vs.vaccination_stance_id
), female_users AS (
		SELECT 
			ROUND(AVG(u.age)) AS "age",
			vs.vaccination_stance_id,
			COUNT(DISTINCT(uvs.user_id)) AS "user_count"
		FROM
			users AS u
		JOIN
			user_vaccination_stances AS uvs ON
			uvs.user_id = u.user_id
		JOIN
			vaccination_stances AS vs ON
			vs.vaccination_stance_id = uvs.vaccination_stance_id
		WHERE
			u.gender = 'Woman'
		GROUP BY
			vs.vaccination_stance_id
)
SELECT 
	vs.name,
	male_users.age AS "Average Male User Age",
	male_users.user_count AS "Male User Count",
	female_users.age AS "Average Female User Age",
	female_users.user_count AS "Female User Count"
FROM 
	vaccination_stances AS vs
JOIN male_users ON
	male_users.vaccination_stance_id = vs.vaccination_stance_id
JOIN female_users ON
	female_users.vaccination_stance_id = vs.vaccination_stance_id;
	
--- (2.4) MOST COMMON POLITICAL ORIENTATION PER COUNTRY
/*This query finds the most common political orientation for each country. It joins
users with user_political_orientations and political_orientations. The function MODE() is used to return the most frequent political orientation within each country. Finally, the result is ordered by the number of users per country.*/
SELECT 
	u.country as "Country",
	MODE() WITHIN GROUP (ORDER BY po.name) as "Most Common Political Orientation",
	COUNT(DISTINCT(upo.user_id)) AS "User Count"
FROM 
	user_political_orientations AS upo
JOIN 
	political_orientations AS po ON 
	upo.political_orientation_id = po.political_orientation_id
JOIN 
	users AS u ON 
	upo.user_id = u.user_id
GROUP BY 
	"Country"
ORDER BY 
	"User Count" DESC;

--- (2.5) MOST COMMON VACCINATION STANCE PER COUNTRY
/*This query finds the most common vaccination stance for each country. It joins users with user_vaccination_stances and 
vaccination_stances. We use again the MODE() function which returns the most frequent vaccination stance within each country. The result will be ordered in descending order*/
SELECT 
	u.country as "Country",
	MODE() WITHIN GROUP (ORDER BY vs.name) as "Most Common Stance on Vaccination", 
	COUNT(DISTINCT(uvs.user_id)) AS "User Count"
FROM 
	user_vaccination_stances AS uvs
JOIN 
	vaccination_stances AS vs ON 
	uvs.vaccination_stance_id = vs.vaccination_stance_id
JOIN 
	users AS u ON 
	uvs.user_id = u.user_id
GROUP BY 
	"Country"
ORDER BY 
	"User Count" DESC;


--- (2.6) INTERESTS SHARED BY USERS FROM MINIMUM 5 COUNTRIES
/*This query finds interest that appear among users from at least five different countries. It will join user_interests with interests to get the interest names and joins users to accessthe country of each user. COUNT(DISTINCT u.country) counts how many different countries are associated with each interest. The HAVING clause keeps only interests appearing in 5 or more countries*/
SELECT 
	i.name as interest, 
	COUNT(DISTINCT u.country) AS num_countries
	FROM user_interests ui
	JOIN interests i on ui.interest_id=i.interest_id
	JOIN users u ON ui.user_id = u.user_id
	WHERE u.country IS NOT NULL
	GROUP BY i.name
	HAVING COUNT(DISTINCT u.country) >= 5
	ORDER BY num_countries DESC;

--- (2.7) MOST COMMON INTERESTS
/*This 7th query finds the most common interests overall. It joins user_interest with interests to get the interest names. Then it will group by interest and counts how many time each interest appears. The final result is ordered from the most frequen interest to the least frequent*/
SELECT 
	i.name AS interest, COUNT(*) AS user_count
	FROM user_interests ui
	JOIN interests i ON ui.interest_id = i.interest_id
	GROUP BY i.name
	ORDER BY user_count DESC;


--- (2.8) TOP 10 GROUPS WITH HIGHEST USER COUNT
/*This query returns the top 10 groups with the highest number of users. It joins groups with users_groups to count how many users belong to each group. The GROUP BY clause groups the results by group and LIMIT 10 keeps the only top 10 groups with the largest user count*/
SELECT 
	g.name AS group_name, COUNT(ug.user_id) AS num_users
	FROM groups g
	JOIN user_groups ug ON g.group_id = ug.group_id
	GROUP BY g.group_id, g.name
	ORDER BY num_users DESC
	LIMIT 10;

--- (2.9) USERS WHO DO NOT BELONG TO ANY GROUP
/*This query finds users who do not belong to any group. It uses NOT EXISTS to check if there is a matching row in user_groups for each user. In the subquery SELECT 1 is used because EXISTS only checks whether a row exists (we do not necessarily need the value returned)*/
SELECT 
	u.user_id, u.nickname, u.country
	FROM users u
	WHERE NOT EXISTS (
	    SELECT 1
	    FROM user_groups ug
	    WHERE ug.user_id = u.user_id
	);


--- (2.10) USERS WITH HIGHER GROUP COUNT THAN AVG
/*This query finds the users who belong to more groups than the average user. First it counts the number of groups for each user. Then the subquery computes the average number of groups per user. The HAVING clause compares each user's group count with that average and keeps only users above the average*/
SELECT 
	u.user_id, u.nickname, COUNT(ug.group_id) AS num_groups
	FROM users u
	JOIN user_groups ug ON u.user_id = ug.user_id
	GROUP BY u.user_id, u.nickname
	HAVING COUNT(ug.group_id) > (
	    SELECT AVG(group_count)
	    FROM (
	        SELECT COUNT(*) AS group_count
	        FROM user_groups
	        GROUP BY user_id
	    ) sub
	)
	ORDER BY num_groups DESC;
